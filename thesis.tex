% \paperchapter{title}{authors}{abstract-input-path}
%   Special "paper-style" chapter: standard \chapter heading, then the
%   author list (italic, centred) and the abstract (sans-serif, indented).
%   Tune the typography here.
\newcommand{\paperchapter}[3]{%
  \chapter{#1}%
  \begin{center}\itshape #2\end{center}%
  \begingroup
    \small\sffamily
    \begin{quote}
      \noindent\textbf{Abstract.}\enspace\input{#3}%
    \end{quote}
  \endgroup
  \par\vspace{0.5\baselineskip}%
}
